package com.abnamro.crms.lpdrisksignaladministration.model;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;
import org.mockito.InjectMocks;
import org.mockito.junit.jupiter.MockitoExtension;

import static org.junit.jupiter.api.Assertions.*;

@ExtendWith(MockitoExtension.class)
class CreditFacilityRMTypeTest {

    @InjectMocks
    private CreditFacilityRMType creditFacilityRMType;

    @BeforeEach
    void setUp() {
        creditFacilityRMType = new CreditFacilityRMType();
        creditFacilityRMType.setCreditFacilityRMTypeId(1L);
        creditFacilityRMType.setFacilityId("FAC123");
        creditFacilityRMType.setRiskManagementType("TypeA");
    }

    @Test
    void testGetCreditFacilityRMTypeId() {
        assertEquals(1L, creditFacilityRMType.getCreditFacilityRMTypeId());
    }

    @Test
    void testSetCreditFacilityRMTypeId() {
        creditFacilityRMType.setCreditFacilityRMTypeId(2L);
        assertEquals(2L, creditFacilityRMType.getCreditFacilityRMTypeId());
    }

    @Test
    void testGetFacilityId() {
        assertEquals("FAC123", creditFacilityRMType.getFacilityId());
    }

    @Test
    void testSetFacilityId() {
        creditFacilityRMType.setFacilityId("FAC456");
        assertEquals("FAC456", creditFacilityRMType.getFacilityId());
    }

    @Test
    void testGetRiskManagementType() {
        assertEquals("TypeA", creditFacilityRMType.getRiskManagementType());
    }

    @Test
    void testSetRiskManagementType() {
        creditFacilityRMType.setRiskManagementType("TypeB");
        assertEquals("TypeB", creditFacilityRMType.getRiskManagementType());
    }
}
