import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertNotNull;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.InjectMocks;
import org.mockito.junit.jupiter.MockitoExtension;
import org.junit.jupiter.api.extension.ExtendWith;

@ExtendWith(MockitoExtension.class)
public class CurrentAccountDTOTest {

    @InjectMocks
    private CurrentAccountDTO currentAccountDTO;

    @BeforeEach
    void setUp() {
        currentAccountDTO = new CurrentAccountDTO();
    }

    @Test
    void testArrangementIdentifier() {
        String arrangementIdentifier = "12345";
        currentAccountDTO.setArrangementIdentifier(arrangementIdentifier);
        assertEquals(arrangementIdentifier, currentAccountDTO.getArrangementIdentifier());
    }

    @Test
    void testCreditFacility() {
        String creditFacility = "Credit Facility";
        currentAccountDTO.setCreditFacility(creditFacility);
        assertEquals(creditFacility, currentAccountDTO.getCreditFacility());
    }

    @Test
    void testExposure() {
        ExposuresDTO exposure = new ExposuresDTO();
        currentAccountDTO.setExposure(exposure);
        assertNotNull(currentAccountDTO.getExposure());
    }
}
