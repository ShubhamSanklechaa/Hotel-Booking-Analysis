@Override
public void saveLPDRiskSignalIntl(List<CreditFacilityDTO> creditFacilityList, List<CurrentAccountDTO> currentAccountsList, String lpdMessage) throws LPDException, InterruptedException, NamingException {
    List<CreditFacilityDTO> nplFacilities = creditFacilityList.stream()
        .filter(facility -> "NPL".equals(facility.getFacilityType()))
        .collect(Collectors.toList());

    if (!nplFacilities.isEmpty()) {
        // Proceed with the NPL facilities for LPD 31 triggers
        generate31Triggers(nplFacilities, lpdMessage);
    } else {
        log.info("No NPL facilities found for trigger generation.");
    }
}
