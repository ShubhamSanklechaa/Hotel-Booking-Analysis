import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.Test;

class ExposuresDTOTest {

    @Test
    void testGettersAndSetters() {
        ExposuresDTO exposuresDTO = new ExposuresDTO();
        exposuresDTO.setGlobalFacilityId("12345");

        assertEquals("12345", exposuresDTO.getGlobalFacilityId());
    }

    @Test
    void testNotNull() {
        ExposuresDTO exposuresDTO = new ExposuresDTO();
        exposuresDTO.setGlobalFacilityId("12345");

        assertNotNull(exposuresDTO.getGlobalFacilityId());
    }

    @Test
    void testToString() {
        ExposuresDTO exposuresDTO = new ExposuresDTO();
        exposuresDTO.setGlobalFacilityId("12345");

        String expected = "ExposuresDTO(globalFacilityId=12345)";
        assertEquals(expected, exposuresDTO.toString());
    }

    @Test
    void testEqualsAndHashCode() {
        ExposuresDTO dto1 = new ExposuresDTO();
        dto1.setGlobalFacilityId("12345");

        ExposuresDTO dto2 = new ExposuresDTO();
        dto2.setGlobalFacilityId("12345");

        assertEquals(dto1, dto2);
        assertEquals(dto1.hashCode(), dto2.hashCode());
    }
}
